# Blog Autopilot Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build an autonomous blog content engine that generates, translates (8 languages), and publishes SEO-optimized posts using Claude Sonnet 4.6, integrated as a cron job in the existing Express server.

**Architecture:** A topic engine selects themes from WhatsApp chatbot data, page visits analytics, and Claude's niche knowledge. A generation pipeline creates articles with knowledge base context, self-reviews them, runs SEO audit, translates to 8 languages, fetches Unsplash images, and saves as drafts. A separate cron auto-publishes mature drafts after a configurable delay. Everything is managed via 7 new MCP tools added to the existing content-server.

**Tech Stack:** TypeScript, Drizzle ORM, Neon PostgreSQL, node-cron (already installed), @anthropic-ai/sdk (new), unsplash-js (new), MCP SDK (already installed)

## Task 3: Blog Topic Engine

**Files:**

This service is responsible for selecting the next topic to write about. It queries WhatsApp conversation data, page visit analytics, existing posts, and cluster status, then uses Claude to propose and rank topics.

**Step 1: Create the topic engine**

import Anthropic from "@anthropic-ai/sdk";

interface TopicSuggestion {

/**
 * Gather context from WhatsApp chatbot conversations to identify
 * frequently asked questions and popular topics.
 */
async function getWhatsappInsights(): Promise<string> {
  try {
    // Get recent conversation contexts (last 90 days)
    const ninetyDaysAgo = new Date();
    ninetyDaysAgo.setDate(ninetyDaysAgo.getDate() - 90);

    const conversations = await db
      .select({
        context: schema.chatbotConversations.context,
        currentState: schema.chatbotConversations.currentState,
        messagesCount: schema.chatbotConversations.messagesCount,
      })
      .from(schema.chatbotConversations)
      .where(gte(schema.chatbotConversations.lastMessageAt, ninetyDaysAgo))
      .orderBy(desc(schema.chatbotConversations.lastMessageAt))
      .limit(100);

    if (conversations.length === 0) return "No WhatsApp conversation data available.";

    // Summarize conversation states (what topics people ask about)
    const stateCounts: Record<string, number> = {};
    for (const conv of conversations) {
      const state = conv.currentState;
      stateCounts[state] = (stateCounts[state] || 0) + 1;
    }

    return `WhatsApp chatbot data (last 90 days, ${conversations.length} conversations):\n` +
      `Conversation states: ${JSON.stringify(stateCounts)}\n` +
      `This indicates what customers most frequently ask about.`;
  } catch {
    return "WhatsApp data unavailable.";
  }
}

/**
 * Analyze page visits to identify high-performing content topics.
 */
async function getPageVisitsInsights(): Promise<string> {
  try {
    const thirtyDaysAgo = new Date();
    thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);

    const topPages = await db
      .select({
        pagePath: schema.pageVisits.pagePath,
        visits: count(),
      })
      .from(schema.pageVisits)
      .where(
        and(
          gte(schema.pageVisits.visitedAt, thirtyDaysAgo),
          sql`${schema.pageVisits.pagePath} LIKE '/blog%'`
        )
      )
      .groupBy(schema.pageVisits.pagePath)
      .orderBy(desc(count()))
      .limit(20);

    if (topPages.length === 0) return "No blog page visit data available yet.";

    return `Top blog pages by visits (last 30 days):\n` +
      topPages.map(p => `  ${p.pagePath}: ${p.visits} visits`).join("\n");
  } catch {
    return "Page visits data unavailable.";
  }
}

/**
 * Get existing blog posts to avoid duplicate topics.
 */
async function getExistingPostsSummary(): Promise<string> {
  const posts = await db
    .select({
      title: schema.blogPosts.title,
      slug: schema.blogPosts.slug,
      category: schema.blogPosts.category,
      tags: schema.blogPosts.tags,
      isPublished: schema.blogPosts.isPublished,
    })
    .from(schema.blogPosts)
    .orderBy(desc(schema.blogPosts.createdAt));

  if (posts.length === 0) return "No existing blog posts.";

  return `Existing blog posts (${posts.length} total):\n` +
    posts.map(p => `  - [${p.isPublished ? "published" : "draft"}] "${p.title}" (/${p.slug}) [${p.category}] tags: ${p.tags?.join(", ") || "none"}`).join("\n");
}

/**
 * Get existing clusters and their status.
 */
async function getClustersSummary(): Promise<string> {
  const clusters = await db
    .select()
    .from(schema.blogClusters)
    .orderBy(desc(schema.blogClusters.createdAt));

  if (clusters.length === 0) return "No clusters created yet. The system should create the first topical cluster.";

  return `Existing clusters:\n` +
    clusters.map(c => {
      const planned = (c.plannedTopics || []).length;
      return `  - "${c.name}" (${c.completedCount}/${planned} completed, ${c.isComplete ? "complete" : "active"}) keywords: ${c.keywords?.join(", ") || "none"}`;
    }).join("\n");
}

/**
 * Find a post that needs refreshing (>30 days old, low traffic).
 */
export async function findPostToRefresh(): Promise<{ postId: string; title: string; slug: string } | null> {
  const thirtyDaysAgo = new Date();
  thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);

  // Get posts older than 30 days
  const oldPosts = await db
    .select({
      id: schema.blogPosts.id,
      title: schema.blogPosts.title,
      slug: schema.blogPosts.slug,
    })
    .from(schema.blogPosts)
    .where(
      and(
        eq(schema.blogPosts.isPublished, true),
        sql`${schema.blogPosts.createdAt} < ${thirtyDaysAgo}`
      )
    );

  if (oldPosts.length === 0) return null;

  // For each post, count visits in last 30 days
  const postsWithVisits = await Promise.all(
    oldPosts.map(async (post) => {
      const [result] = await db
        .select({ visits: count() })
        .from(schema.pageVisits)
        .where(
          and(
            sql`${schema.pageVisits.pagePath} = ${`/blog/${post.slug}`}`,
            gte(schema.pageVisits.visitedAt, thirtyDaysAgo)
          )
        );
      return { ...post, visits: result?.visits ?? 0 };
    })
  );

  // Sort by least visits and return the worst performer
  postsWithVisits.sort((a, b) => Number(a.visits) - Number(b.visits));
  const worst = postsWithVisits[0];

  return worst ? { postId: worst.id, title: worst.title, slug: worst.slug } : null;
}

): Promise<{ improved: GeneratedArticle; tokensIn: number; tokensOut: number }> {
  const response = await client.messages.create({
    model,
    max_tokens: 4096,
    messages: [
      {
        role: "user",
        content: `You are a senior editor reviewing a blog article for a boat rental business.

Review and improve this article. Fix any issues with:
- SEO: keyword density, heading structure, meta description length
- Quality: readability, flow, factual accuracy
- Engagement: hook, CTA, internal linking
- Grammar and style

ORIGINAL ARTICLE:
${JSON.stringify(article, null, 2)}

Return the IMPROVED version in the exact same JSON format. If the article is already excellent, return it unchanged. Do not wrap in markdown.`,
      },
    ],
  });

  const text = response.content[0].type === "text" ? response.content[0].text : "";
  const improved: GeneratedArticle = JSON.parse(text);

  return {
    improved,
    tokensIn: response.usage.input_tokens,
    tokensOut: response.usage.output_tokens,
  };
}

/**
 * Step 3: Run SEO audit on the generated content (mirrors get_seo_audit logic).
 */
function runSeoAudit(article: GeneratedArticle): { score: number; issues: string[] } {
  const issues: string[] = [];
  let score = 100;

  // Title check
  if (!article.title || article.title.length < 30) {
    issues.push(`Title too short (${article.title?.length || 0} chars)`);
    score -= 10;
  } else if (article.title.length > 70) {
    issues.push(`Title too long (${article.title.length} chars)`);
    score -= 5;
  }

  // Meta description
  if (!article.metaDescription || article.metaDescription.length < 120) {
    issues.push(`Meta description too short (${article.metaDescription?.length || 0} chars)`);
    score -= 10;
  } else if (article.metaDescription.length > 160) {
    issues.push(`Meta description too long (${article.metaDescription.length} chars)`);
    score -= 5;
  }

  // Content length
  const wordCount = article.content.split(/\s+/).filter(Boolean).length;
  if (wordCount < 300) {
    issues.push(`Content too thin (${wordCount} words)`);
    score -= 25;
  } else if (wordCount < 600) {
    issues.push(`Content short (${wordCount} words)`);
    score -= 10;
  }

  // Headings
  if (!/^#{2,3}\s/m.test(article.content)) {
    issues.push("No H2/H3 headings found");
    score -= 10;
  }

  // Internal links
  if (!/\[.*?\]\(\/.*?\)/.test(article.content)) {
    issues.push("No internal links found");
    score -= 10;
  }

  // Excerpt
  if (!article.excerpt || article.excerpt.length < 20) {
    issues.push("Missing or too short excerpt");
    score -= 5;
  }

  // Tags
  if (!article.tags || article.tags.length < 3) {
    issues.push("Less than 3 tags");
    score -= 3;
  }

  return { score: Math.max(0, score), issues };
}

/**
 * Step 4: Translate the article to all configured languages.
 */
async function translateArticle(
  client: Anthropic,
  article: GeneratedArticle,
  targetLanguages: string[],
  model: string
): Promise<{
  titleByLang: Record<string, string>;
  contentByLang: Record<string, string>;
  excerptByLang: Record<string, string>;
  metaDescByLang: Record<string, string>;
  tokensIn: number;
  tokensOut: number;
}> {
  const titleByLang: Record<string, string> = { es: article.title };
  const contentByLang: Record<string, string> = { es: article.content };
  const excerptByLang: Record<string, string> = { es: article.excerpt };
  const metaDescByLang: Record<string, string> = { es: article.metaDescription };

  let totalTokensIn = 0;
  let totalTokensOut = 0;

  // Filter out Spanish (already have it) and translate to remaining languages
  const langsToTranslate = targetLanguages.filter(l => l !== "es");

  if (langsToTranslate.length === 0) {
    return { titleByLang, contentByLang, excerptByLang, metaDescByLang, tokensIn: 0, tokensOut: 0 };
  }

  // Translate in batches of 3 to avoid rate limits
  const BATCH_SIZE = 3;
  for (let i = 0; i < langsToTranslate.length; i += BATCH_SIZE) {
    const batch = langsToTranslate.slice(i, i + BATCH_SIZE);

    const results = await Promise.all(
      batch.map(async (lang) => {
        const langNames: Record<string, string> = {
          en: "English", fr: "French", de: "German",
          it: "Italian", nl: "Dutch", ru: "Russian", ca: "Catalan",
        };

        const response = await client.messages.create({
          model,
          max_tokens: 4096,
          messages: [
            {
              role: "user",
              content: `Translate this blog article to ${langNames[lang] || lang}. Adapt it naturally (not word-for-word). Keep Markdown formatting, internal links, and proper nouns unchanged.

Respond in this JSON format (no markdown wrapping):
{
  "title": "Translated title",
  "content": "Translated full Markdown content",
  "excerpt": "Translated excerpt",
  "metaDescription": "Translated meta description (max 160 chars)"
}

ARTICLE TO TRANSLATE:
Title: ${article.title}
Excerpt: ${article.excerpt}
Meta Description: ${article.metaDescription}
Content:
${article.content}`,
            },
          ],
        });

        const text = response.content[0].type === "text" ? response.content[0].text : "";
        const translated = JSON.parse(text);

        return {
          lang,
          translated,
          tokensIn: response.usage.input_tokens,
          tokensOut: response.usage.output_tokens,
        };
      })
    );

    for (const { lang, translated, tokensIn, tokensOut } of results) {
      titleByLang[lang] = translated.title;
      contentByLang[lang] = translated.content;
      excerptByLang[lang] = translated.excerpt;
      metaDescByLang[lang] = translated.metaDescription;
      totalTokensIn += tokensIn;
      totalTokensOut += tokensOut;
    }
  }

  return {
    titleByLang,
    contentByLang,
    excerptByLang,
    metaDescByLang,
    tokensIn: totalTokensIn,
    tokensOut: totalTokensOut,
  };
}

/**
 * Step 5: Fetch a relevant image from Unsplash.
 */
async function fetchUnsplashImage(keywords: string[]): Promise<string | null> {
  const unsplash = getUnsplashClient();
  if (!unsplash) return null;

  try {
    const query = keywords.slice(0, 3).join(" ") + " boat sea";
    const result = await unsplash.search.getPhotos({
      query,
      perPage: 1,
      orientation: "landscape",
    });

    if (result.response && result.response.results.length > 0) {
      const photo = result.response.results[0];
      // Use regular size (1080px wide) and add UTM params per Unsplash guidelines
      return `${photo.urls.regular}&utm_source=costabravarentaboat&utm_medium=referral`;
    }
  } catch (error) {
    console.error("[BlogAutopilot] Unsplash fetch error:", error);
  }

  return null;
}

/**
 * Run the full autopilot pipeline: topic → generate → review → audit → translate → save.
 */
export async function runAutopilotPipeline(
  options: { forceGeneration?: boolean; type?: "new" | "refresh" | "manual" } = {}
): Promise<AutopilotRunResult> {
  const type = options.type || "new";

  try {
    const client = getAnthropicClient();
    if (!client) {
      return { success: false, error: "ANTHROPIC_API_KEY not configured", type };
    }

    const config = await getConfig();
    const apiKey = process.env.ANTHROPIC_API_KEY!;

    // Season check (skip for manual runs)
    if (!options.forceGeneration && !isInSeason(config)) {
      return { success: false, error: "Off-season. Use forceGeneration to override.", type };
    }

    // Weekly limit check
    if (!options.forceGeneration) {
      const postsThisWeek = await getPostsThisWeek();
      if (postsThisWeek >= config.maxPostsPerWeek) {
        return { success: false, error: `Weekly limit reached (${postsThisWeek}/${config.maxPostsPerWeek})`, type };
      }
    }

    let totalTokensIn = 0;
    let totalTokensOut = 0;

    // Decide: new post or refresh?
    const runCount = await db
      .select({ count: count() })
      .from(schema.blogAutopilotLog)
      .then(r => Number(r[0]?.count ?? 0));

    const shouldRefresh = type === "refresh" ||
      (type === "new" && config.refreshRatio > 0 && runCount > 0 && runCount % config.refreshRatio === 0);

    if (shouldRefresh && type !== "manual") {
      const postToRefresh = await findPostToRefresh();
      if (postToRefresh) {
        console.log(`[BlogAutopilot] Refreshing post: "${postToRefresh.title}"`);
        // For refresh, we re-generate with the same topic
        return await runRefreshPipeline(client, config, postToRefresh);
      }
    }

    // Step 1: Select topic
    console.log("[BlogAutopilot] Selecting topic...");
    const topic = await selectNextTopic(apiKey);
    console.log(`[BlogAutopilot] Topic selected: "${topic.topic}"`);

    // Step 2: Get business context
    const businessContext = await getBusinessContext();
    const clusterContext = `This article is part of the "${topic.clusterName}" cluster (type: ${topic.type}).`;

    // Step 3: Generate article
    console.log("[BlogAutopilot] Generating article...");
    const { article, tokensIn: genIn, tokensOut: genOut } = await generateArticle(
      client, topic.topic, topic.keywords, topic.category, clusterContext, businessContext, config.model
    );
    totalTokensIn += genIn;
    totalTokensOut += genOut;

    // Step 4: Self-review
    console.log("[BlogAutopilot] Self-reviewing...");
    const { improved, tokensIn: revIn, tokensOut: revOut } = await selfReviewArticle(client, article, config.model);
    totalTokensIn += revIn;
    totalTokensOut += revOut;

    // Step 5: SEO audit
    let seoResult = runSeoAudit(improved);
    console.log(`[BlogAutopilot] SEO score: ${seoResult.score}`);

    // Retry if score too low (max 1 retry)
    let finalArticle = improved;
    if (seoResult.score < config.minSeoScore) {
      console.log(`[BlogAutopilot] Score below ${config.minSeoScore}, regenerating...`);
      const { article: retry, tokensIn: retryIn, tokensOut: retryOut } = await generateArticle(
        client, topic.topic, topic.keywords, topic.category, clusterContext, businessContext, config.model
      );
      totalTokensIn += retryIn;
      totalTokensOut += retryOut;

      const { improved: retryImproved, tokensIn: retryRevIn, tokensOut: retryRevOut } = await selfReviewArticle(client, retry, config.model);
      totalTokensIn += retryRevIn;
      totalTokensOut += retryRevOut;

      seoResult = runSeoAudit(retryImproved);
      finalArticle = retryImproved;
      console.log(`[BlogAutopilot] Retry SEO score: ${seoResult.score}`);
    }

    // Step 6: Translate
    console.log("[BlogAutopilot] Translating to", config.languages.length, "languages...");
    const translations = await translateArticle(client, finalArticle, config.languages, config.model);
    totalTokensIn += translations.tokensIn;
    totalTokensOut += translations.tokensOut;

    // Step 7: Fetch image
    let featuredImage: string | null = null;
    if (config.unsplashEnabled) {
      console.log("[BlogAutopilot] Fetching image...");
      featuredImage = await fetchUnsplashImage(finalArticle.keywords);
    }

    // Step 8: Ensure cluster exists
    let clusterId: string | null = null;
    const [existingCluster] = await db
      .select()
      .from(schema.blogClusters)
      .where(eq(schema.blogClusters.name, topic.clusterName))
      .limit(1);

    if (existingCluster) {
      clusterId = existingCluster.id;
    } else {
      const [newCluster] = await db
        .insert(schema.blogClusters)
        .values({
          name: topic.clusterName,
          keywords: topic.keywords,
          plannedTopics: [{ topic: topic.topic, type: topic.type }],
        })
        .returning();
      clusterId = newCluster.id;
    }

    // Step 9: Save as draft (will be auto-published after delay)
    const [post] = await db
      .insert(schema.blogPosts)
      .values({
        title: finalArticle.title,
        slug: finalArticle.slug,
        content: finalArticle.content,
        excerpt: finalArticle.excerpt,
        metaDescription: finalArticle.metaDescription,
        category: finalArticle.category,
        tags: finalArticle.tags,
        author: "Costa Brava Rent a Boat",
        featuredImage,
        isPublished: false, // Draft until delay expires
        titleByLang: translations.titleByLang,
        contentByLang: translations.contentByLang,
        excerptByLang: translations.excerptByLang,
        metaDescByLang: translations.metaDescByLang,
        clusterId,
        isAutoGenerated: true,
        seoScore: seoResult.score,
      })
      .returning();

    // Update cluster if this was a pillar post
    if (topic.type === "pillar" && clusterId) {
      await db
        .update(schema.blogClusters)
        .set({ pillarPostId: post.id, updatedAt: new Date() })
        .where(eq(schema.blogClusters.id, clusterId));
    }

    // Increment cluster completed count
    if (clusterId) {
      await db
        .update(schema.blogClusters)
        .set({
          completedCount: sql`${schema.blogClusters.completedCount} + 1`,
          updatedAt: new Date(),
        })
        .where(eq(schema.blogClusters.id, clusterId));
    }

    // Step 10: Log execution
    await db.insert(schema.blogAutopilotLog).values({
      postId: post.id,
      type,
      topicChosen: topic.topic,
      clusterName: topic.clusterName,
      keywordsUsed: topic.keywords,
      modelUsed: config.model,
      tokensInput: totalTokensIn,
      tokensOutput: totalTokensOut,
      seoScore: seoResult.score,
      status: "success",
    });

    console.log(`[BlogAutopilot] Post created: "${finalArticle.title}" (score: ${seoResult.score}, ID: ${post.id})`);

    return {
      success: true,
      postId: post.id,
      topic: topic.topic,
      seoScore: seoResult.score,
      tokensInput: totalTokensIn,
      tokensOutput: totalTokensOut,
      type,
    };
  } catch (error) {
    const message = error instanceof Error ? error.message : String(error);
    console.error("[BlogAutopilot] Pipeline error:", message);

    // Log the error
    await db.insert(schema.blogAutopilotLog).values({
      type,
      status: "error",
      errorMessage: message,
    }).catch(() => {}); // Don't fail on log error

    return { success: false, error: message, type };
  }
}

/**
 * Refresh an existing post with new/improved content.
 */
async function runRefreshPipeline(
  client: Anthropic,
  config: schema.BlogAutopilotConfig,
  post: { postId: string; title: string; slug: string }
): Promise<AutopilotRunResult> {
  try {
    const [existing] = await db
      .select()
      .from(schema.blogPosts)
      .where(eq(schema.blogPosts.id, post.postId));

    if (!existing) {
      return { success: false, error: `Post ${post.postId} not found`, type: "refresh" };
    }

    const businessContext = await getBusinessContext();

    const response = await client.messages.create({
      model: config.model,
      max_tokens: 4096,
      messages: [
        {
          role: "user",
          content: `You are an SEO editor. This blog post needs refreshing to improve its search ranking.

CURRENT ARTICLE:
Title: ${existing.title}
Category: ${existing.category}
Content: ${existing.content}

${businessContext}

Rewrite and improve this article:
- Update and expand the content (aim for 800-1500 words)
- Improve keyword optimization
- Add/update internal links
- Improve headings and structure
- Keep the same general topic and slug

Respond in JSON (no markdown wrapping):
{
  "title": "Improved title",
  "content": "Improved full Markdown content",
  "excerpt": "Updated excerpt",
  "metaDescription": "Updated meta description (max 160 chars)",
  "tags": ["tag1", "tag2", "tag3"]
}`,
        },
      ],
    });

    const text = response.content[0].type === "text" ? response.content[0].text : "";
    const refreshed = JSON.parse(text);

    const seoResult = runSeoAudit({
      ...refreshed,
      slug: existing.slug,
      category: existing.category,
      keywords: refreshed.tags,
    });

    // Translate refreshed content
    const translations = await translateArticle(
      client,
      { ...refreshed, slug: existing.slug, category: existing.category, keywords: refreshed.tags },
      config.languages,
      config.model
    );

    // Update the post
    await db
      .update(schema.blogPosts)
      .set({
        title: refreshed.title,
        content: refreshed.content,
        excerpt: refreshed.excerpt,
        metaDescription: refreshed.metaDescription,
        tags: refreshed.tags,
        titleByLang: translations.titleByLang,
        contentByLang: translations.contentByLang,
        excerptByLang: translations.excerptByLang,
        metaDescByLang: translations.metaDescByLang,
        seoScore: seoResult.score,
        updatedAt: new Date(),
      })
      .where(eq(schema.blogPosts.id, post.postId));

    // Log
    await db.insert(schema.blogAutopilotLog).values({
      postId: post.postId,
      type: "refresh",
      topicChosen: `Refresh: ${existing.title}`,
      modelUsed: config.model,
      tokensInput: response.usage.input_tokens + translations.tokensIn,
      tokensOutput: response.usage.output_tokens + translations.tokensOut,
      seoScore: seoResult.score,
      status: "success",
    });

    console.log(`[BlogAutopilot] Post refreshed: "${refreshed.title}" (score: ${seoResult.score})`);

    return {
      success: true,
      postId: post.postId,
      topic: `Refresh: ${existing.title}`,
      seoScore: seoResult.score,
      tokensInput: response.usage.input_tokens + translations.tokensIn,
      tokensOutput: response.usage.output_tokens + translations.tokensOut,
      type: "refresh",
    };
  } catch (error) {
    const message = error instanceof Error ? error.message : String(error);
    return { success: false, error: message, type: "refresh" };
  }
}

/**
 * Auto-publish drafts that have passed the publish delay.
 */
export async function publishMatureDrafts(): Promise<number> {
  const config = await getConfig();
  const cutoff = new Date();
  cutoff.setHours(cutoff.getHours() - config.publishDelayHours);

  const drafts = await db
    .select()
    .from(schema.blogPosts)
    .where(
      and(
        eq(schema.blogPosts.isPublished, false),
        eq(schema.blogPosts.isAutoGenerated, true),
        sql`${schema.blogPosts.createdAt} <= ${cutoff}`
      )
    );

  let published = 0;
  for (const draft of drafts) {
    await db
      .update(schema.blogPosts)
      .set({
        isPublished: true,
        publishedAt: new Date(),
        updatedAt: new Date(),
      })
      .where(eq(schema.blogPosts.id, draft.id));
    published++;
    console.log(`[BlogAutopilot] Auto-published: "${draft.title}"`);
  }

  return published;
}

server.tool(
  "autopilot_queue",
  "View or manage the topic queue. Can list planned topics, regenerate the queue, or discard a topic.",
  {
    action: z.enum(["list", "regenerate", "discard"]).describe("Action to perform"),
    topicId: z.string().optional().describe("Topic ID to discard (required for 'discard' action)"),
  },
  async ({ action, topicId }) => {
    try {
      if (action === "list") {
        const queue = await db
          .select()
          .from(schema.blogAutopilotQueue)
          .where(eq(schema.blogAutopilotQueue.status, "planned"))
          .orderBy(schema.blogAutopilotQueue.priority);

        return {
          content: [{
            type: "text" as const,
            text: JSON.stringify({ count: queue.length, topics: queue }, null, 2),
          }],
        };
      }

      if (action === "regenerate") {
        const apiKey = process.env.ANTHROPIC_API_KEY;
        if (!apiKey) {
          return { content: [{ type: "text" as const, text: "ANTHROPIC_API_KEY not set" }], isError: true };
        }

        // Mark old planned topics as discarded
        await db
          .update(schema.blogAutopilotQueue)
          .set({ status: "discarded" })
          .where(eq(schema.blogAutopilotQueue.status, "planned"));

        // Generate new queue
        const topics = await generateTopicQueue(apiKey, 6);

        const inserted = [];
        for (let i = 0; i < topics.length; i++) {
          const t = topics[i];
          const [row] = await db
            .insert(schema.blogAutopilotQueue)
            .values({
              topic: t.topic,
              keywords: t.keywords,
              type: t.type,
              priority: i + 1,
              clusterId: null,
            })
            .returning();
          inserted.push(row);
        }

        return {
          content: [{
            type: "text" as const,
            text: JSON.stringify({ success: true, message: `Generated ${inserted.length} topics`, topics: inserted }, null, 2),
          }],
        };
      }

      if (action === "discard" && topicId) {
        await db
          .update(schema.blogAutopilotQueue)
          .set({ status: "discarded" })
          .where(eq(schema.blogAutopilotQueue.id, topicId));

        return {
          content: [{ type: "text" as const, text: JSON.stringify({ success: true, message: `Topic ${topicId} discarded` }) }],
        };
      }

      return { content: [{ type: "text" as const, text: "Invalid action or missing topicId" }], isError: true };
    } catch (error: unknown) {
      const message = error instanceof Error ? error.message : String(error);
      return { content: [{ type: "text" as const, text: `Error: ${message}` }], isError: true };
    }
  }
);
